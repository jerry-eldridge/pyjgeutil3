START START_PROG -> PROGRAM_START
PROGRAM_START SEMICOLON -> BODY
BODY WHILE -> BOOLEAN_COND
BODY NAME -> BODY_STATEMENT_ASSIGN
BODY_STATEMENT_ASSIGN ASSIGN -> BODY_STATEMENT_EXPR
BODY_STATEMENT_ASSIGN BEGIN -> BODY
BODY_STATEMENT_EXPR NAME -> BODY_STATEMENT_PLUS
BODY_STATEMENT_PLUS PLUS -> BODY_STATEMENT_NUM
BODY_STATEMENT_NUM NUM -> BODY_END_STATEMENT
BODY PRINT -> BODY_STATEMENT_PRINT
BODY_STATEMENT_PRINT NAME -> BODY_STRING
BODY_STATEMENT_PRINT QUOTE -> BODY_STRING
BODY_STRING STRING -> BODY_STRING
BODY_STRING QUOTE -> BODY_STRING
BODY_STRING SEMICOLON -> BODY
BODY_END_STATEMENT SEMICOLON -> BODY
BODY IF -> BOOLEAN_COND
BOOLEAN BEGIN -> BODY
BODY BEGIN -> BODY
BODY END -> BODY
BODY ELSE -> BODY
BOOLEAN_COND NAME -> BOOLEAN_COND_OP
BOOLEAN_COND_OP LT -> BOOLEAN_COND_VAL
BOOLEAN_COND_OP GT -> BOOLEAN_COND_VAL
BOOLEAN_COND_OP LE -> BOOLEAN_COND_VAL
BOOLEAN_COND_OP GE -> BOOLEAN_COND_VAL
BOOLEAN_COND_OP EQ -> BOOLEAN_COND_VAL
BOOLEAN_COND_VAL NUM -> BOOLEAN
BODY FINISH -> FINISH_SEMICOLON
FINISH_SEMICOLON SEMICOLON -> HALT
